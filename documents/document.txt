database name BRGY_MAYPANDAN

CREATE TABLE users (
    id SERIAL PRIMARY KEY NOT NULL,
    fName VARCHAR(255) NOT NULL,
    mName VARCHAR(255) NULL,
    lName VARCHAR(255) NOT NULL,
    username VARCHAR(255) NOT NULL,
    password VARCHAR(255) NOT NULL
);

CREATE TABLE categories (
    categoryId SERIAL PRIMARY KEY NOT NULL,
    categoryName VARCHAR(255) NOT NULL
);

CREATE TABLE inventory (
    id SERIAL PRIMARY KEY NOT NULL,
    iName VARCHAR(255) NOT NULL,  
    quantity INT DEFAULT 0,
    iPrice INT NOT NULL,
    dateAdded DATE NOT NULL,
    categoryId INT REFERENCES categories(categoryId) NOT NULL,
    isFunctional BOOLEAN DEFAULT TRUE
);

CREATE TABLE archive (
    id SERIAL PRIMARY KEY NOT NULL,
    name VARCHAR(255) NOT NULL,
    date DATE NOT NULL,
    img TEXT NOT NULL,
    docType VARCHAR(255) NOT NULL
);

-- residents classification
CREATE TABLE rClassification (
    rClassificationId SERIAL PRIMARY KEY NOT NULL,
    rClassificationName VARCHAR(255) NOT NULL
);

-- in case of emergency contact person 
CREATE TABLE contactPerson(
     contactPersonId SERIAL PRIMARY KEY NOT NULL,
     fName VARCHAR(255) NOT NULL,         -31
     mName VARCHAR(255) NULL,             -32
     lName VARCHAR(255) NOT NULL,         -33
     street VARCHAR(255) NULL,            -34
     purok VARCHAR(255) NULL,             -35
     barangay VARCHAR(255) NULL,          -36
     city VARCHAR(255) NULL,              -37
     province VARCHAR(255) NULL,          -38
     contactNumber VARCHAR(50) NULL       -39
);

// non-residents
CREATE TABLE boarders (
    boardID SERIAL PRIMARY KEY NOT NULL,
    boarderInResidentId INT REFERENCES residents(residentsId) NOT NULL,
    originalStreet VARCHAR(255) NULL,      -24                                                -- used as a original address of boarder
    originalPurok VARCHAR(255) NULL,       -25                                              -- used as a original address of boarder
    originalBarangay VARCHAR(255) NULL,    -26                                               -- used as a original address of boarder
    originalCity VARCHAR(255) NULL,        -27                                             -- used as a original address of boarder
    originalProvince VARCHAR(255) NULL,    -28                                             -- used as a original address of boarder
    boardinghouseName VARCHAR(255) NULL,   -29
    landlord VARCHAR(255) NULL             -30
);

CREATE TABLE residents (
    residentsId SERIAL PRIMARY KEY NOT NULL,
    globalId VARCHAR(255) NOT NULL UNIQUE, 
    idNumber VARCHAR(255) NULL, 
    fName VARCHAR(255) NOT NULL,           
    mName VARCHAR(255) NULL,               
    lName VARCHAR(255) NOT NULL,           
    street VARCHAR(255) NULL,                                                    -- used as a address of boarder while staying in the barangay
    purok VARCHAR(255) NULL,                                                     -- used as a address of boarder while staying in the barangay
    barangay VARCHAR(255) NULL,                                                  -- used as a address of boarder while staying in the barangay
    city VARCHAR(255) NULL,                                                      -- used as a address of boarder while staying in the barangay
    province VARCHAR(255) NULL,                                                  -- used as a address of boarder while staying in the barangay
    birthDate DATE NOT NULL,                
    birthPlace VARCHAR(255) NULL,           
    age INT NOT NULL,                       
    gender VARCHAR(10) NULL CHECK (gender IN ('Male', 'Female')),  
    picture TEXT DEFAULT '',                                       
    signature TEXT DEFAULT '',  -- reserve only if needed
    eAttainment VARCHAR(255) NULL,                                 
    occupation VARCHAR(255) NULL,                                  
    income INT NULL,                                               
    civilStatus VARCHAR(50) NULL CHECK (civilStatus IN ('Single', 'Married', 'Divorced')),   
    isResident BOOLEAN NULL,                                                                 
    emergencyContactId INT REFERENCES contactPerson(contactPersonId) NOT NULL,       
    rClassificationId INT REFERENCES rClassification(rClassificationId) NOT NULL,   -- not available is boarder
    isPwd BOOLEAN DEFAULT FALSE,                                                 -- not available if boarder
    isSoloParent BOOLEAN DEFAULT FALSE,                                          -- not available if boarder
    isYouth BOOLEAN DEFAULT FALSE,                                               -- not available if boarder
    is4ps BOOLEAN DEFAULT FALSE                                                  -- not available if boarder
);

CREATE TABLE requests (
    id SERIAL PRIMARY KEY NOT NULL,
    residentsId INT REFERENCES residents(residentsId) NOT NULL,    
    dateAdded DATE NULL,
    purpose TEXT NULL,
    isReleased BOOLEAN DEFAULT FALSE
);
